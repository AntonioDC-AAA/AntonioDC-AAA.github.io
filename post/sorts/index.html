冒泡排序

每次外循环都能确定数组的最大值

func BubbleSort(nums []int){
	if nums == nil || len(nums) < 2{
		return
	}
	for i:=len(nums)-1;i>=0;i--{
		for j:=0;j<i;j++{
			if nums[j] > nums[j+1]{
				nums[j],nums[j+1] = nums[j+1],nums[j]
			}
		}
	}
	fmt.Println(nums)
}

选择排序

每次外循环确定数组的最小值

func SelectSort(nums []int){
	if nums == nil || len(nums) < 2{
		return
	}
	for i:=0;i<len(nums)-1;i++{
		for j:=i+1;j<len(nums);j++{
			if nums[i] > nums[j]{
				nums[i],nums[j] = nums[j],nums[i]
			}
		}
	}
	fmt.Println(nums)
}

}

func MergeSort(nums []int, left int, right int) {
	if left == right {
		return
	}
	mid := left + (right-left)/2
	MergeSort(nums, left, mid)			
	MergeSort(nums, mid+1, right)
	merge(nums, left, right, mid)
}
//外排序
func merge(nums []int, left int, right int, mid int) {
	var ans []int
	ans = make([]int, right-left+1)
	p1 := 0
	p2 := mid + 1
	p3 := left
	fmt.Println("the len is ", len(ans))
	for p3 <= mid && p2 <= right {
		if nums[p3] < nums[p2] {
			ans[p1] = nums[p3]
			p3++
			p1++
		} else {
			ans[p1] = nums[p2]
			p2++
			p1++
		}
	}

	for p3 <= mid {
		ans[p1] = nums[p3]
		p3++
		p1++
	}
	for p2 <= right {
		ans[p1] = nums[p2]
		p2++
		p1++
	}

	for i := 0; i < len(ans); i++ {
		nums[left+i] = ans[i]
	}
}

//荷兰国旗的先导   本质是双指针
func padding(nums []int, target int) {
	var t int

	for i := 0; i < len(nums); i++ {
		if nums[i] < target {
			nums[t], nums[i] = nums[i], nums[t]
			t++
		}
	}

	fmt.Println(nums)
}

//荷兰国旗
func flagHolland(nums []int, target int) {
	var loc int
	var more int = len(nums) - 1
	var less int
	for loc <= more {
		if nums[loc] < target {
			nums[loc], nums[less] = nums[less], nums[loc]
			loc++
			less++
		} else if nums[loc] > target {
			nums[loc], nums[more] = nums[more], nums[loc]
			more--
		} else {
			loc++
		}
	}
}

经典快排

一个数组最后一个元素为x

把小于等于x的放左边，大于x的放在右边

递归这个过程

修改后的快排

把小于x的放左边，大于x的放在右边，等于x的放在中间（荷兰国旗思想）

这样数据量到达一定程度后会变成冒泡排序

每次都是根据后面的元素进行快排的，加入随机因子，每次都是选择随机下标

在工程上不允许递归函数

func QuickSort(nums []int) {
	if nums == nil || len(nums) < 2 {
		return
	}

	quickSort(nums, 0, len(nums)-1)
}

func quickSort(nums []int, start int, end int) {
	if start < end {
		se := quickSortGetStartEnd(nums, start, end)
		fmt.Println(se)
		quickSort(nums, start, se[0]-1)
		quickSort(nums, se[1]+1, end-1)
	}
}


func quickSortGetStartEnd(nums []int, start int, end int) []int {
	var p int
	var target int = nums[end]

	for p <= end {
		fmt.Println(nums[p], nums[end])
		if nums[p] < target { //这个地方不能是nums[end]，因为每次end的时候都会变化
			nums[p], nums[start] = nums[start], nums[p]
			p++
			start++
		} else if nums[p] > target {
			nums[p], nums[end] = nums[end], nums[p]
			end--
		} else {
			p++
		}
	}
	return []int{start, end}
}
